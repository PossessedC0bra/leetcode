import {getAllPages} from 'nextra/context';
import {Card, Cards} from 'nextra/components'

export function ProblemIndex({children}) {
    const pages = getAllPages();

    return <>
        <Cards num={1}>
            {pages && pages
                .filter(page => !page.meta.excludeFromIndex && page.kind === 'MdxPage')
                .map(page =>
                    <Card title={page.meta.title} href={page.route}/>
                )
            }
        </Cards>
    </>

}
